2) Service Worker 註冊建議用相對路徑（GitHub Pages: /travel-plan/）：
if ('serviceWorker' in navigator) {
  window.addEventListener('load', () => {
    navigator.serviceWorker.register('./sw.js', { scope: './' });
  });
}
